99 lines of code on the wall

Hello, my name is Austin Lee. Firstly, let me finish that song... 

linesOutput = -> ( num, wall: true ) do
  wallOutput = if wall
                 " on the wall"
               end
  "#{num} line#{ num != 1 ? 's' : '' } of code#{wallOutput}"
end

99.downto(1).each do |numOfLines|
  puts linesOutput[numOfLines]
  puts linesOutput[numOfLines, wall: false]
  puts "Take one down, refactor around!"
  if numOfLines > 1
    puts linesOutput[numOfLines - 1]
  else
    puts "No more lines of code on the wall!"
  end
end
